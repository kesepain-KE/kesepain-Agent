# - start.py

# - provider/provider.py
# - provider/LLM/openai_api.py
#
# Explicitly excluded from this root file:
# - system/skills/**
# - system/mcp/**

# - provider/LLM/google_api.py (placeholder, not wired into the main chain)

# Rule:
# - Skill-specific dependencies do not register here.
# - If a future skill needs third-party packages, document them with that skill

#   main runtime chain.
