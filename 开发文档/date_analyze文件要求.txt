[start模式]
1.接收传参:
type:start
2.启动core\chat.py，传参：
type:start
3.等待数据返回：
type:runed
4.启动core\memory.py，传参：
type:read
5.等待数据返回：
type:runed
6.启动core\agent.py，传参：
type:start

[continue模式]
1.接收传参：
type:continue
2.启动core\memory.py，传参：
type:continue
3.等待参数返回:
type:runed
4.启动core\chat.py，传参：
type:continue
5.等待数据返回：
type:runed
6.启动core\memory.py，传参：
type:read
7.等待数据返回：
type:runed
8.启动core\agent.py，传参：
type:continue
